项目根目录新建 vue.config.js

# 配置别名

const path = require('path')
const resolve = dir => path.join(__dirname, dir)
module.exports = {
  chainWebpack: config => {
    config.resolve.alias
      .set('@', resolve('./src'))
      .set('assets', resolve('./src/assets'))
      .set('com', resolve('./src/components'))
      .set('views', resolve('./src/views'))
  }
}
