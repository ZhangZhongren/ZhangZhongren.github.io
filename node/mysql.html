# brew 安装mysql

  1. 安装
 brew install mysql
  1. 启动服务
  mysql.server start
  1. 执行安全设置并且设置密码
  mysql_secure_installation
  1. 终端登陆
  mysql -u root -p
  1. 创建数据库
  create database {name} character set utf8mb4;
  1. 创建新用户
  create user '{name}' identified by '{pwd}';
  1. 授权用户
  grant all privileges on retail_db.* to '{name}';
  1. 查看数据库
  show databases;
  1. 显示当前数据库的表单
  show tables
  1. 退出终端 mysql
quit;
  1. 停止 mysql 服务
  mysql.server stop
