# 区分单击双击事件

TIP

利用延时器 setTimeout 和清除定时器方法 clearTimeout

demo


let ODiv = document.getElementById('id')
let timer = 0
ODiv.onclick = () => {
  clearTimeout(timer)
  timer = setTimeout(() => {
    // do somethind
  }, 200)
}

ODiv.ondblclick = () => {
  clearTimeout(timer)
  // do somethind
}
